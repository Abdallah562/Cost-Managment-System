Cost Management System

1. Cost Management

GET /api/CostManagement/GetAll

Description: Retrieve all cost entries.
Response:

[
    {
        "id": 1,
        "description": "Cloud hosting fees",
        "amount": 350.75,
        "date": "2024-11-12T09:00:00",
        "category": "IT Services"
    }
]

POST /api/CostManagement/AddEntry

Description: Create a new cost entry.
Request:

{
    "description": "Team building event",
    "amount": 580.00,
    "category": "HR"
}

Response:

{
    "id": 1,
    "description": "Team building event",
    "amount": 580.00,
    "date": "2024-11-13T14:45:00",
    "category": "HR"
}

}

2. Invoice Management

GET /api/Invoice

Description: Retrieve all invoices.
Response:

[
    {
        "invoiceId": 1,
        "clientName": "Acme Corp",
        "subtotal": 900.00,
        "tax": 45.00,
        "discount": 30.00,
        "grandTotal": 915.00,
        "items": [
            {
	            "id":  1,
                "name": "Software Licensing",
                "unitPrice": 450.00,
                "quantity": 2,
                "invoiceId": 1
            }
        ]
    }
]

GET /api/Invoice/{invoiceId}

Description: Retrieve specific invoice by ID.
Response:

{
    "invoiceId": 1,
    "clientName": "Acme Corp",
    "subtotal": 900.00,
    "tax": 45.00,
    "discount": 30.00,
    "grandTotal": 915.00,
    "items": [
        {
	        "id":  1,
            "name": "Software Licensing",
            "unitPrice": 450.00,
            "quantity": 2,
            "invoiceId": 1
        }
    ]
}

POST /api/Invoice/Generate

Description: Create new invoice.
Request:

{
  "clientId": 2045,
  "tax": 15,
  "discount": 10,
  "items": [
    {
          "name": "Software Licensing",
          "unitPrice": 450.00,
          "quantity": 2
    }
  ]
}

Response:

{
	"id":  5,
	"clientId":  2045,
	"tax":  15,
	"discount":  10,
	"items":  [
		{
		"id":  9,
		"name":  "Software Licensing",
		"quantity":  2,
		"unitPrice":  450,
		"invoiceId":  5
		}
	]
}

PUT /api/Invoice/{invoiceId}/Edit

Description: Update existing invoice.
Request:

[
	{
		"name":  "Premium Support Plan",
		"unitPrice":  600.00,
		"quantity":  1
	}
]

Response:

{
	"id":  1,
	"clientId":  2,
	"tax":  14,
	"discount":  10,
	"items":  [
		{
			"id":  10,
			"name":  "Premium Support Plan",
			"quantity":  1,
			"unitPrice":  600,
			"invoiceId":  1
		}
	]
}

POST /api/Invoice/SendReminder

Description: Send payment reminder.
Request:

{
    "invoiceId": 1,
    "dueDate": "2025-07-01",
    "clientEmail": "test@example.com"
}

Response:

{

	"clientEmail":  "test@example.com",
	"subject":  "Reminder: Invoice 1 is due soon",
	"body":  "Dear client, your invoice with Id : 1 with a total price of 16800 is due in 2 days on July 01, 2025. Please make payment on time."

}

3. Tax

POST /api/Tax/Calculate

Description: Calculate taxes.
Request:

{
    "subtotal": 2500,
    "taxRate": 20,
    "region": "UK"
}

Response:

{
	"taxAmount":  500.0,
	"totalWithTax":  3000.0
}
